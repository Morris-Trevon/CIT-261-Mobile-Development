Properties, Methods, Object Creation Functions

function checkTime(){

// Array with months to convert numeral months
var months = ["January", "February", "March", "April", "May", "June",
"July", "August", "September", "October", "November", "December"];

//Create a date object using the object Date Constructor method. You can use this method to create any object eg. [var user = new Object();]

//For this example we are using the Date object which has many properties in function format
var today = new Date();

//Create an Object called time
var time = {
//Properties of the Object. We are also Using the getSeconds(), getMinute(), getMonth(), getFullYear(), etc methods
//by appending to today Date object so we can get the correct date
year : today.getFullYear(),
day : today.getDay(),
month : today.getMonth(),
hours : today.getHours(),
//Functions of the Object will add a 0 before single digit numbers for uniformity
//Using object functions allow for complex calculations to be done within the object
minutes : function(){

if(today.getMinutes() < 10) {
return "0"+today.getMinutes();
}else{
return today.getMinutes();
}

},
seconds : function(){

if(today.getSeconds() < 10) {

return "0"+today.getSeconds();
}else{
return today.getSeconds();
}

},
milliSeconds : today.getMilliseconds()

};

//We now print out the time object

document.getElementById('output').innerHTML = "Time: "+months[time.month]+" "+time.day+", "+time.year+", "+time.hours+":"+time.minutes()+":"+time.seconds();
//To tell the time in realtime we use the setTimeout method which takes two parameters setTimeout([functionName],[resetTime]

var timeReset = setTimeout(checkTime, 500);
//We can also return the time object to be used by another object if necessary while resetting this function without any issues
return time;
}

function unRavelStorage2(){
//declare variables in the same manner as before
var carID = document.getElementById('carID2').value;
var carColorPicker2 = document.getElementById("carColorPicker2");
var carColor = carColorPicker2.options[carColorPicker2.selectedIndex].text;

//pull the item out of Storage in the same manner
var getCar = localStorage.getItem(carID);
var retrievedCar = JSON.parse(getCar);
//validate just the same
if (carID === "") {
alert("Please Enter the Car ID");
return false;
}


//Instantiate an Object with Prototype method instead and here is where things get different
var carColorHolder = function(carModel, carColor, carID) {

//Create an empty object using Object.create()
var car = Object.create(carColorHolder.prototype);

//Create the properties needed to collect data
//object.key = value
car.carModel = carModel;
car.carColor = carColor;
car.carID = carID;

return car;
}

//The carColorHolder.prototype can be used outside the object to create functions
carColorHolder.prototype.whatModel = function(){
return this.carModel;
}

//You can now call the object like a function and pass values to it the same as before
var finalCarOrder2 = carColorHolder(retrievedCar.carModel,carColor,carID);

//If successful the values should now be in finalCarOrder2 lets use the METHOD toUpperCase to display color this time.
document.getElementById('output4').innerHTML = "<br><br>"+"Your "+finalCarOrder2.carColor.toUpperCase()+" "+finalCarOrder2.whatModel()+" with carID of: "+finalCarOrder2.carID+"<br>Is ready for pickup!";

// Our prototype object works and all is well
return finalCarOrder2;
}
